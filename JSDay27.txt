1.Write a program to filter a list of names. Use textbox, and ul. On typing the letters in the textbox the list should be filtered.

<!DOCTYPE html>
<html lang="en">
<head>
    <title>Filter Names</title>
</head>
<body>

    <h2>Filter Names</h2>

    <input type="text" id="searchBox" placeholder="Type to search...">

    <ul id="nameList">
        <li>Dinesh</li>
        <li>Arun</li>
        <li>Priya</li>
        <li>Karthik</li>
        <li>Divya</li>
    </ul>

    <script>
        const searchBox = document.getElementById("searchBox");
        const names = document.querySelectorAll("#nameList li");

        searchBox.addEventListener("input", function () {
            const value = this.value.toLowerCase();

            names.forEach(function (name) {
                if (name.textContent.toLowerCase().includes(value)) {
                    name.style.display = "block";
                } else {
                    name.style.display = "none";
                }
            });
        });
    </script>

</body>
</html>
2.Display a list of cards containing movie name and image using javascript. 

<!DOCTYPE html>
<html lang="en">
<head>
    <title>Movie Cards</title>
    <style>
        .container {
            display: flex;
            gap: 20px;
            flex-wrap: wrap;
        }

        .card {
            width: 150px;
            border: 1px solid #ccc;
            padding: 10px;
            text-align: center;
            border-radius: 10px;
        }

        .card img {
            width: 100%;
            height: 200px;
            object-fit: cover;
        }
    </style>
</head>
<body>

    <h2>Movie List</h2>
    <div class="container" id="movieContainer"></div>

    <script>
        const movies = [
            {
                name: "Leo",
                image: "https://via.placeholder.com/150"
            },
            {
                name: "Jailer",
                image: "https://via.placeholder.com/150"
            },
            {
                name: "Vikram",
                image: "https://via.placeholder.com/150"
            }
        ];

        const container = document.getElementById("movieContainer");

        movies.forEach(movie => {
            const card = document.createElement("div");
            card.classList.add("card");

            card.innerHTML = `
                <img src="${movie.image}" alt="${movie.name}">
                <h3>${movie.name}</h3>
            `;

            container.appendChild(card);
        });
    </script>

</body>
</html>
