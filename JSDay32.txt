// Promise code using async/await

function myPromise() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve("Promise Success");
        }, 2000);
    });
}

async function displayData() {
    let result = await myPromise();
    console.log(result);
}

displayData();
